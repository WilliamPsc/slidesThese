\begin{frame}{Threat model}
    \begin{block}{Threat model}
        \begin{itemize}
            \item<1> DIFT-related registers + protection-related registers
            \item<2> Single bit-flip in two registers at two distinct clock cycles% -> 1 fault / clock cycle
            \item<3> Single bit-flip in two registers at a given clock cycle% -> 2 faults / clock cycle
            \item<4> Multi-bit faults in one register at a given clock cycle% -> 6 faults / clock cycle
            \item<5> \textbf{\underline{Multi-bit faults in two registers at a given clock cycle}} {\tiny (registers from 1 to 10 bits only)}% -> 11 faults / clock cycle
        \end{itemize}
    \end{block}
\end{frame}
